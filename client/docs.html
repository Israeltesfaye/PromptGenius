Endpoint 1

Description

This endpoint does something.

Request


POST /api/endpoint-1
{
"key1": "value1",
"key2": "value2"
}

Response


{
"success": true,
"data": {
"key1": "value1",
"key2": "value2"
}
}

Example Code


const fetch = require('node-fetch');

async function callEndpoint1() {
const response = await fetch('https://example.com/api/endpoint-1', {
method: 'POST',
headers: {
'Content-Type': 'application/json'
},
body: JSON.stringify({
key1: 'value1',
key2: 'value2'
})
});

const data = await response.json();

console.log(data);
}

callEndpoint1();
